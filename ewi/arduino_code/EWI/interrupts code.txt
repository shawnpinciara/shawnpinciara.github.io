  nota = nota | mask1; 
  MIDI.sendNoteOn(scale[noteIndex[int(nota)]+noteIndex[int(chiave)]-2]+octave, velocity, 1);  // Send a MIDI note on message
}

  MIDI.sendNoteOn(scale[noteIndex[int(nota)]+noteIndex[int(chiave)]-2]+octave, velocity, 1);  // Send a MIDI note on message
}
